⚠ multiset: ms.erase(x) ลบทุกตัวที่เท่ากับ x! ลบตัวเดียว → ms.erase(ms.find(x)) ⚠ ลบระหว่างวน: it = s.erase(it); (erase คืน iterator ตัวถัดไป) — ห้าม s.erase(it); it++; ⚠ ห้ามแก้ค่าที่อยู่ใน set (มันจะไม่เรียงใหม่) — ลบออกแล้ว insert ตัวใหม่แทน multiset.count(x) ช้าถ้าซ้ำเยอะ (O(log n + จำนวนซ้ำ)) — ปกติไม่มีปัญหา

Binary Search (จุดตาย: ปัดขึ้น/ปัดลง)

// แบบ 1: หา x "น้อยสุด" ที่ ok(x) จริง   (หน้าตา F F F T T T)
ll lo = ต่ำสุดที่เป็นไปได้, hi = สูงสุดที่ ok แน่ๆ;
while (lo < hi) {
    ll mid = lo + (hi - lo) / 2;         // ปัดลง
    if (ok(mid)) hi = mid;
    else lo = mid + 1;
}
// จบ: lo == hi == คำตอบ

// แบบ 2: หา x "ใหญ่สุด" ที่ ok(x) จริง   (หน้าตา T T T F F F)
while (lo < hi) {
    ll mid = lo + (hi - lo + 1) / 2;     // ⚠ ปัดขึ้น! ใช้ปัดลง = loop ค้างไม่จบ
    if (ok(mid)) lo = mid;
    else hi = mid - 1;
}

Binary search on answer — สัญญาณ: โจทย์ถาม "น้อยสุด/มากสุดเท่าไหร่ที่ทำได้" และการเช็ค "ค่า t นี้ทำได้ไหม" ง่ายกว่าหาคำตอบตรงๆ → เขียน ok(t) แล้วครอบด้วยแบบ 1 หรือ 2

  • ok ต้อง monotone: จริงแล้วจริงตลอดทาง (เช็คในหัวก่อนใช้)
  • ใน ok(t) ระวังผลรวม overflow → สะสมเป็น ll และ break เมื่อเกินเป้าแล้ว
  • (lo + hi) / 2 ตรงๆ overflow ได้ถ้าทั้งคู่ใหญ่ → ใช้ lo + (hi - lo) / 2

Two Pointers / Prefix Sum (บรรทัดเสี่ยง)

// sliding window: subarray ผลรวม = x — ใช้ได้เมื่อเลข "บวกล้วน" เท่านั้น
int l = 0; ll sum = 0, cnt = 0;
for (int r = 0; r < n; r++) {
    sum += a[r];
    while (sum > x) sum -= a[l++];
    if (sum == x) cnt++;
}

// มีเลขติดลบ → sliding window ใช้ไม่ได้ ต้อง prefix + map:
map<ll,ll> seen; seen[0] = 1;      // ⚠ seed prefix ว่าง — ลืม = ผิดทุกเคสที่ prefix == x
ll pre = 0, cnt = 0;
for (int i = 0; i < n; i++) {
    pre += a[i];
    cnt += seen[pre - x];
    seen[pre]++;
}

DSU (Union-Find): "อยู่กลุ่มเดียวกันไหม" เกือบ O(1)

vector<int> par, sz;
void dsu_init(int n) {
    par.resize(n + 1); sz.assign(n + 1, 1);
    iota(par.begin(), par.end(), 0);            // par[i] = i
}
int find(int x) {
    return par[x] == x ? x : par[x] = find(par[x]);
}
bool uni(int a, int b) {                        // รวมกลุ่ม a,b — คืน false ถ้ากลุ่มเดียวกันอยู่แล้ว
    a = find(a); b = find(b);
    if (a == b) return false;
    if (sz[a] < sz[b]) swap(a, b);
    par[b] = a; sz[a] += sz[b];
    return true;
}

วิธีใช้ (ไม่ต้องเข้าใจไส้ใน): - อยู่กลุ่มเดียวกันไหม → find(a) == find(b) - ขนาดกลุ่มของ a → sz[find(a)] - นับจำนวนกลุ่ม: เริ่ม = n, ทุกครั้งที่ uni() คืน true → ลด 1 - edge ไหนทำให้เกิดวงกลม → uni(u,v) คืน false = edge นั้นสร้าง cycle - สัญญาณใช้: "เพิ่มถนน/สายไฟทีละเส้น แล้วถามว่าเชื่อมถึงกันไหม" ระหว่างทาง (BFS ตอบตอนจบได้ แต่ตอบระหว่างเพิ่มทีละเส้นไม่ไหว)

        dp[w] = max(dp[w], dp[w - wt[i]] + val[i]);
// คำตอบ: dp[W]

// ทำไมถอยหลัง: เดินหน้า = dp[w-wt] อาจถูกชิ้น "นี้" อัปเดตไปแล้วในรอบเดียวกัน = หยิบซ้ำ
// เดินหน้า = หยิบซ้ำได้ไม่จำกัด (unbounded) — คือ Coin Comb II ที่เคยเขียนนั่นเอง

// subset sum: เซตตัวเลข สร้าง "ผลรวม" อะไรได้บ้าง (เช่น Money Sums)
vector<char> can(X + 1, 0); can[0] = 1;
for (ll c : a)
    for (int s = X; s >= c; s--)            // ถอยหลังเหมือนกัน (ตัวละครั้ง)
        if (can[s - c]) can[s] = 1;
// ตอบ: ทุก s ที่ can[s] == 1
// "แบ่งสองกองให้ใกล้กันสุด" = subset sum แล้วหา s ใกล้ ผลรวม/2 ที่ can[s]

สัญญาณ: "เลือกของภายใต้งบ/น้ำหนักจำกัด" • "สร้างค่า x ได้ไหม" • "แบ่งของสองกอง" ⚠ ใช้ได้เมื่อ W/X ≤ ~1e6 (dp ตามขนาดผลรวม ไม่ใช่จำนวนชิ้น)

LCS / Edit Distance / LIS

ต่อยอดจาก: 2D DP (Array Description) — state สองมิติ i,j = ตำแหน่งในสอง string

// LCS: ลำดับย่อยร่วมยาวสุดของ a, b (ไม่ต้องติดกัน)  O(nm) — n,m ≤ ~5000
vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
for (int i = 1; i <= n; i++)
    for (int j = 1; j <= m; j++)
        dp[i][j] = (a[i-1] == b[j-1])              // ⚠ string index = i-1 (dp เลื่อน 1)
                 ? dp[i-1][j-1] + 1
                 : max(dp[i-1][j], dp[i][j-1]);
// คำตอบ: dp[n][m]

// edit distance (แก้ a เป็น b น้อยครั้งสุด: ลบ/เพิ่ม/แทนที่) — โครงเดียวกัน:
// dp[i][0] = i, dp[0][j] = j (base!)
// ตรงกัน → dp[i-1][j-1]  ไม่ตรง → 1 + min({dp[i-1][j-1], dp[i-1][j], dp[i][j-1]})

// LIS: ลำดับย่อยเพิ่มขึ้นยาวสุด  O(n log n) — แค่ lower_bound ที่ใช้เป็นอยู่แล้ว
vector<ll> t;                      // t[k] = ตัวปลายเล็กสุดของลำดับยาว k+1
for (ll x : a) {
    auto it = lower_bound(t.begin(), t.end(), x);
    if (it == t.end()) t.push_back(x);
    else *it = x;
}
// คำตอบ = t.size()
// เพิ่มขึ้น "แท้" (< เท่านั้น) → lower_bound   เท่ากันได้ (≤) → เปลี่ยนเป็น upper_bound
// ⚠ t ไม่ใช่ลำดับจริง (ความยาวถูกอย่างเดียว) — โจทย์ขอตัวลำดับ = คนละเรื่อง

ชุดขยาย 1 — Tree & Range Queries (Fenwick, SegTree, LCA)

ต่อยอดจาก: Prefix Sum (หน้า 12) & BFS/DFS — สำหรับช่วงที่มีการอัปเดตและค้นหาเร็ว O(log N)

1. Fenwick Tree (BIT) — 1-indexed (ขนาด N+1, O(log N)) เขียนง่ายและเร็วกว่า SegTree มาก

struct BIT {
    int n;
    vector<ll> tree;
    BIT(int n) : n(n), tree(n + 1, 0) {}
    void add(int i, ll delta) {
        for (; i <= n; i += i & -i) tree[i] += delta;
    }
    ll query(int i) {
        ll sum = 0;
        for (; i > 0; i -= i & -i) sum += tree[i];
        return sum;
    }
    ll range_query(int l, int r) { return query(r) - query(l - 1); }
};
// Range Update / Point Query: update [l, r] เพิ่ม x -> bit.add(l, x); bit.add(r + 1, -x); อ่านค่าที่ i -> bit.query(i)

2. Segment Tree (0-indexed array, 1-indexed tree node 1..4N) — O(log N)

struct SegTree {
    int n;
    vector<ll> tree;
    SegTree(const vector<ll>& a) {
        n = a.size();
        tree.assign(4 * n, 0);
        build(a, 1, 0, n - 1);
    }
    void build(const vector<ll>& a, int node, int l, int r) {
        if (l == r) { tree[node] = a[l]; return; }
        int mid = (l + r) / 2;
        build(a, 2 * node, l, mid);
        build(a, 2 * node + 1, mid + 1, r);
        tree[node] = tree[2 * node] + tree[2 * node + 1]; // เปลี่ยนเป็น min/max ได้ตามโจทย์
    }
    void update(int node, int l, int r, int idx, ll val) {
        if (l == r) { tree[node] = val; return; } // ถ้าเพิ่มค่าใช้ += val
        int mid = (l + r) / 2;
        if (idx <= mid) update(2 * node, l, mid, idx, val);
        else update(2 * node + 1, mid + 1, r, idx, val);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }
    ll query(int node, int l, int r, int ql, int qr) {
        if (ql <= l && r <= qr) return tree[node];
        if (qr < l || r < ql) return 0; // ถ้าหา min คืน INF, ถ้า max คืน -INF
        int mid = (l + r) / 2;
        return query(2 * node, l, mid, ql, qr) + query(2 * node + 1, mid + 1, r, ql, qr);
    }
};
// เรียกใช้: SegTree st(a); st.query(1, 0, n-1, L, R); st.update(1, 0, n-1, idx, val);

3. Lowest Common Ancestor (LCA via Binary Lifting) — O(N log N) init, O(log N) query

const int LOG = 20; // 2^20 > 1e6
vector<vector<int>> adj_lca, up;
vector<int> depth;
void lca_dfs(int u, int p, int d) {
    depth[u] = d;
    up[u][0] = p;
    for (int k = 1; k < LOG; k++) up[u][k] = up[up[u][k - 1]][k - 1];
    for (int v : adj_lca[u]) if (v != p) lca_dfs(v, u, d + 1);
}
int get_lca(int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);
    for (int k = LOG - 1; k >= 0; k--)
        if (depth[u] - (1 << k) >= depth[v]) u = up[u][k];
    if (u == v) return u;
    for (int k = LOG - 1; k >= 0; k--)
        if (up[u][k] != up[v][k]) { u = up[u][k]; v = up[v][k]; }
    return up[u][0];
}
int dist_lca(int u, int v) { return depth[u] + depth[v] - 2 * depth[get_lca(u, v)]; }
// ก่อนใช้: adj_lca.assign(n + 1, {}); up.assign(n + 1, vector<int>(LOG, 0)); depth.assign(n + 1, 0);
//          lca_dfs(root, root, 0);
// ⚠ lca_dfs เป็น recursion — tree เป็นเส้นตรงยาว ~2e5 เสี่ยง stack overflow

for (int i = 1; i <= n && is_bipartite; i++) {

3. Bridge & Articulation Point (Tarjan's Algorithm) — หาจุดตัดและสะพานขาดใน O(V+E)

vector<int> tin(n + 1, -1), low(n + 1, -1);
int timer_tj = 0;
vector<pair<int,int>> bridges;
vector<char> is_art(n + 1, 0);
void tarjan_dfs(int u, int p = -1) {
    tin[u] = low[u] = timer_tj++;
    int children = 0;
    for (int v : adj[u]) {
        if (v == p) continue;
        if (tin[v] != -1) { low[u] = min(low[u], tin[v]); } // back-edge
        else {
            tarjan_dfs(v, u); // tree-edge
            low[u] = min(low[u], low[v]);
            if (low[v] > tin[u]) bridges.push_back({u, v}); // Bridge (เส้นเดียวระหว่าง u-v ตัดแล้วขาด)
            if (low[v] >= tin[u] && p != -1) is_art[u] = 1;  // Articulation Point
            children++;
        }
    }
    if (p == -1 && children > 1) is_art[u] = 1; // Root เป็นจุดตัดถ้ามีลูก > 1 กิ่ง
}
// เรียก: วนทุก u ที่ tin[u] == -1 → tarjan_dfs(u)   (กราฟอาจมีหลาย component)
// ⚠ ถ้ามี edge ซ้ำ (u-v สองเส้น) จะตอบ bridge ผิด — ต้องเปลี่ยนไป skip ด้วย "index ของ edge" แทน parent
// ⚠ recursion ลึก ~2e5 เสี่ยง stack overflow

ชุดขยาย 3 — String Algorithms (Z-Algo, KMP, Rolling Hash)

ต่อยอดจาก: string (หน้า 4) — เมื่อ substr/find ธรรมดาช้าเกินไป O(NM)

1. Z-Algorithm — z[i] = ความยาว prefix ของ s ที่ตรงกับ substring ตั้งแต่ i (O(N))

vector<int> z_function(const string& s) {
    int n = s.size(), l = 0, r = 0;
    vector<int> z(n, 0);
    for (int i = 1; i < n; i++) {
        if (i <= r) z[i] = min(r - i + 1, z[i - l]);
        while (i + z[i] < n && s[z[i]] == s[i + z[i]]) z[i]++;
        if (i + z[i] - 1 > r) { l = i; r = i + z[i] - 1; }
    }
    return z;
}
// ค้นหา pattern P ใน text T: สร้าง s = P + "$" + T แล้วหา i ที่ z[i] == P.size()

2. KMP (Knuth-Morris-Pratt) — ค้นหา Pattern P ใน Text T ใน O(N+M)

vector<int> prefix_function(const string& s) {
    int n = s.size();
    vector<int> pi(n, 0);
    for (int i = 1; i < n; i++) {
        int j = pi[i - 1];
        while (j > 0 && s[i] != s[j]) j = pi[j - 1];
        if (s[i] == s[j]) j++;
        pi[i] = j;
    }
    return pi;
}
// การหา: pi จาก P+"$"+T ถ้าตำแหน่งไหนได้ pi[i] == P.size() คือเจอที่ตำแหน่งนั้น

3. Polynomial Rolling Hash — หา hash ของ substring ใน O(1) (ใช้ 2 base กันชน)

struct StringHash {
    const ll BASE1 = 31, MOD1 = 1e9 + 7;
    const ll BASE2 = 37, MOD2 = 1e9 + 9;
    vector<ll> h1, h2, p1, p2;
    StringHash(const string& s) {
        int n = s.size();
        h1.assign(n + 1, 0); h2.assign(n + 1, 0);
        p1.assign(n + 1, 1); p2.assign(n + 1, 1);
        for (int i = 0; i < n; i++) {
            h1[i + 1] = (h1[i] * BASE1 + (s[i] - 'a' + 1)) % MOD1;
            h2[i + 1] = (h2[i] * BASE2 + (s[i] - 'a' + 1)) % MOD2;
            p1[i + 1] = (p1[i] * BASE1) % MOD1;
            p2[i + 1] = (p2[i] * BASE2) % MOD2;
        }
    }
    pair<ll, ll> get_hash(int l, int r) { // 0-indexed [l, r]
        ll res1 = (h1[r + 1] - h1[l] * p1[r - l + 1]) % MOD1;
        if (res1 < 0) res1 += MOD1;
        ll res2 = (h2[r + 1] - h2[l] * p2[r - l + 1]) % MOD2;
        if (res2 < 0) res2 += MOD2;
        return {res1, res2};
    }
};

ชุดขยาย 4 — Math & Number Theory Extensions (Matrix Pow, ExtGCD, SPF)

ต่อยอดจาก: Math (หน้า 17) & DP (หน้า 14)

1. Matrix Exponentiation — คูณเมทริกซ์และยกกำลังแก้โจทย์ Recurrence DP ใน O(K^3 log E)

struct Matrix {
    int r, c;
    vector<vector<ll>> mat;
    Matrix(int r, int c) : r(r), c(c), mat(r, vector<ll>(c, 0)) {}
    Matrix operator*(const Matrix& o) const {
        Matrix res(r, o.c);
        for (int i = 0; i < r; i++)
            for (int k = 0; k < c; k++)
                for (int j = 0; j < o.c; j++)
                    res.mat[i][j] = (res.mat[i][j] + mat[i][k] * o.mat[k][j]) % MOD;
        return res;
    }
};
Matrix matpow(Matrix a, ll e) {
    Matrix res(a.r, a.c);
    for (int i = 0; i < a.r; i++) res.mat[i][i] = 1; // Identity
    while (e > 0) {
        if (e & 1) res = res * a;
        a = a * a;
        e >>= 1;
    }
    return res;
}
// ตัวอย่าง Fibonacci F(n): transition matrix = {{1, 1}, {1, 0}} ยกกำลัง n-1 แล้วคูณ {{F(1)}, {F(0)}}

2. Extended GCD & Modular Inverse (สำหรับ mod ทั่วไปที่ gcd(a, m) == 1)

ll extgcd(ll a, ll b, ll& x, ll& y) {
    if (b == 0) { x = 1; y = 0; return a; }
    ll x1, y1;
    ll g = extgcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - y1 * (a / b);
    return g;
}
ll modinv_general(ll a, ll m) {
    ll x, y;
    ll g = extgcd(a, m, x, y);
    if (g != 1) return -1; // หาไม่ได้ถ้า gcd != 1
    return (x % m + m) % m;
}

3. Smallest Prime Factor (SPF) — แยกตัวประกอบเร็ว O(log N) เมื่อ N ≤ 1e7

vector<int> spf(N + 1);
void sieve_spf() {
    iota(spf.begin(), spf.end(), 0);
    for (int i = 2; i * i <= N; i++)
        if (spf[i] == i)
            for (int j = i * i; j <= N; j += i)
                if (spf[j] == j) spf[j] = i;
}
vector<pair<int, int>> factorize(int x) {
    vector<pair<int, int>> res;
    while (x > 1) {
        int p = spf[x], cnt = 0;
        while (x % p == 0) { cnt++; x /= p; }
        res.push_back({p, cnt});
    }
    return res;
}
// จำนวนตัวหาร = (cnt1 + 1) * (cnt2 + 1) * ...

ชุดขยาย 5 — Geometry Basics & Convex Hull (Monotone Chain)

ต่อยอดจาก: vector & pair (หน้า 3, 8) — สำหรับโจทย์เรขาคณิตและการหาเปลือกนอก

1. 2D Point Struct & Cross/Dot Product Basics

typedef long long T;
struct Point {
    T x, y;
    Point(T x = 0, T y = 0) : x(x), y(y) {}
    Point operator+(const Point& p) const { return Point(x + p.x, y + p.y); }
    Point operator-(const Point& p) const { return Point(x - p.x, y - p.y); }
    T dot(const Point& p) const { return x * p.x + y * p.y; }
    T cross(const Point& p) const { return x * p.y - y * p.x; }
    bool operator<(const Point& p) const { return x < p.x || (x == p.x && y < p.y); }
};
// ทิศทางเลี้ยว o -> a -> b: (a - o).cross(b - o) > 0 คือซ้าย (CCW), < 0 คือขวา (CW), = 0 คือเส้นตรงเดียวกัน
T cross_op(Point o, Point a, Point b) { return (a - o).cross(b - o); }
// พื้นที่รูปหลายเหลี่ยม (Shoelace formula): abs(p[0].cross(p[1]) + p[1].cross(p[2]) + ...) / 2.0
// (ไม่ใส่ abs จะได้ค่าติดลบเมื่อจุดเรียงตามเข็มนาฬิกา)

2. Convex Hull (Monotone Chain) — หาเปลือกนอกใน O(N log N)

vector<Point> convex_hull(vector<Point>& pts) {
    int n = pts.size(), k = 0;
    if (n <= 2) return pts;
    vector<Point> h(2 * n);
    sort(pts.begin(), pts.end());
    for (int i = 0; i < n; i++) {
        while (k >= 2 && cross_op(h[k - 2], h[k - 1], pts[i]) <= 0) k--; // เปลี่ยน <= เป็น < ถ้าเอาจุดบนขอบด้วย
        h[k++] = pts[i];
    }
    for (int i = n - 2, t = k + 1; i >= 0; i--) {
        while (k >= t && cross_op(h[k - 2], h[k - 1], pts[i]) <= 0) k--;
        h[k++] = pts[i];
    }
    h.resize(k - 1);
    return h;
}

ชุดขยาย 6 — Dynamic Programming Extensions (Digit DP, Bitmask DP/TSP, Knapsack Tricks)

ต่อยอดจาก: DP (หน้า 14) — สำหรับโจทย์นับตัวเลขตามเงื่อนไข, กราฟ NP-hard ขนาดเล็ก และ Knapsack ที่ W มหาศาล

1. Digit DP (นับจำนวนเลขในช่วง [0, N] ที่ตรงเงื่อนไข) — O(len · states)

// ตัวอย่าง: นับเลขที่ไม่มีเลข 4 อยู่เลย ในช่วง [0, N]
string s = to_string(N);
vector<vector<ll>> memo(s.size(), vector<ll>(2, -1)); // [pos][tight] (เพิ่มมิติ state ได้ตามโจทย์)
ll digit_dp(int pos, bool tight) {
    if (pos == (int)s.size()) return 1; // จบตัวเลข 1 แบบที่ถูกต้อง
    if (memo[pos][tight] != -1) return memo[pos][tight];
    int limit = tight ? (s[pos] - '0') : 9;
    ll res = 0;
    for (int d = 0; d <= limit; d++) {
        if (d == 4) continue; // เงื่อนไขห้ามมีเลข 4
        res += digit_dp(pos + 1, tight && (d == limit));
    }
    return memo[pos][tight] = res;
}
// คำตอบ [A, B] = digit_dp ของ B ลบ digit_dp ของ (A - 1)
// ⚠ ระหว่างสองครั้งต้อง RESET: สร้าง s = to_string(...) ใหม่ + เติม memo เป็น -1 ใหม่ ไม่งั้นคำตอบผิดเงียบๆ

2. Bitmask DP & Travelling Salesperson Problem (TSP) — O(2^N · N^2) เมื่อ N ≤ 20

// dp[mask][u] = ค่าเดินทางน้อยสุดเมื่อผ่านเซตของเมืองใน mask และจบที่เมือง u
const ll INF_DP = 1e18;
vector<vector<ll>> dp(1 << n, vector<ll>(n, INF_DP));
dp[1][0] = 0; // เริ่มต้นที่เมือง 0 (mask = 1)
for (int mask = 1; mask < (1 << n); mask++) {
    for (int u = 0; u < n; u++) {
        if (!(mask >> u & 1) || dp[mask][u] == INF_DP) continue;
        for (int v = 0; v < n; v++) {
            if (mask >> v & 1) continue; // ไปแล้วข้าม
            int next_mask = mask | (1 << v);
            dp[next_mask][v] = min(dp[next_mask][v], dp[mask][u] + dist[u][v]);
        }
    }
}
// คำตอบ: min ของ dp[(1<<n)-1][u] + dist[u][0] (ถ้าต้องวนกลับ)

3. Knapsack เมื่อ W ใหญ่มาก (W ≤ 1e9) แต่มูลค่า V เล็ก (N·Vmax ≤ 1e5)

// สลับ State: dp[val] = น้ำหนักน้อยที่สุดที่ทำให้ได้มูลค่ารวมเท่ากับ val
int max_val = 100000;
vector<ll> dp_knap(max_val + 1, INF_DP);
dp_knap[0] = 0;
for (int i = 0; i < n; i++)
    for (int v = max_val; v >= val[i]; v--)
        if (dp_knap[v - val[i]] != INF_DP)
            dp_knap[v] = min(dp_knap[v], dp_knap[v - val[i]] + wt[i]);
// คำตอบ: v ใหญ่สุดจาก max_val..0 ที่ dp_knap[v] <= W (ความจุเป้)

ชุดขยาย 7 — Advanced Graph Extensions (2-SAT & Eulerian Path)

ต่อยอดจาก: Toposort & DFS (หน้า 21, 22)

1. 2-SAT (Two-Satisfiability) — หาค่าความจริงเงื่อนไข (X_i ∨ Y_j) ใน O(V+E)

// 2-SAT: ตัวแปร i = โหนด 2*i, NOT i = โหนด 2*i+1 (0-indexed 0..n-1)
struct TwoSat {
    int n;
    vector<vector<int>> adj, radj;
    vector<int> order, comp, ans;
    vector<char> vis;
    TwoSat(int n) : n(n), adj(2*n), radj(2*n), comp(2*n, -1), ans(n, 0), vis(2*n, 0) {}
    void add_clause(int i, bool f, int j, bool g) { // (X_i = f) OR (X_j = g)
        int u = 2*i + (f?0:1), not_u = 2*i + (f?1:0);
        int v = 2*j + (g?0:1), not_v = 2*j + (g?1:0);
        adj[not_u].push_back(v); radj[v].push_back(not_u); // NOT u -> v
        adj[not_v].push_back(u); radj[u].push_back(not_v); // NOT v -> u
    }
    void dfs1(int u) { vis[u]=1; for(int v:adj[u]) if(!vis[v]) dfs1(v); order.push_back(u); }
    void dfs2(int u, int c) { comp[u]=c; for(int v:radj[u]) if(comp[v]==-1) dfs2(v, c); }
    bool solve() {
        for(int i=0; i<2*n; i++) if(!vis[i]) dfs1(i);
        reverse(order.begin(), order.end());
        int c = 0;
        for(int u:order) if(comp[u]==-1) dfs2(u, c++);
        for(int i=0; i<n; i++) {
            if(comp[2*i] == comp[2*i+1]) return false; // ขัดแย้ง เป็นไปไม่ได้!
            ans[i] = comp[2*i] > comp[2*i+1];          // SCC ที่ออกหลังเป็นจริง
        }
        return true;
    }
};
// ⚠ dfs1/dfs2 เป็น recursion ลึกได้ถึง 2n — n ใหญ่เสี่ยง stack overflow

2. Eulerian Path / Circuit (Hierholzer's Algorithm) — ลากเส้นผ่านทุก Edge พอดีครั้งเดียว O(V+E)

// กราฟมีทิศทาง: Circuit มีได้เมื่อ in-deg == out-deg ทุกโหนด
// Path มีได้เมื่อ มีจุดเริ่ม 1 จุด (out - in == 1), จุดจบ 1 จุด (in - out == 1) และที่เหลือ in == out
vector<vector<int>> adj_eu; // เก็บปลายทาง (หรือ index ของ edge)
vector<int> path;
void euler_dfs(int u) {
    while (!adj_eu[u].empty()) {
        int v = adj_eu[u].back();
        adj_eu[u].pop_back();
        euler_dfs(v);
    }
    path.push_back(u); // ใส่ตอนถอยกลับ
}
// คำตอบ: euler_dfs(start_node); reverse(path.begin(), path.end());
// ⚠ ต้องเช็คก่อนว่า edge ทุกเส้นเชื่อมถึงกัน (component เดียว) ไม่งั้นได้ path ไม่ครบแต่ไม่ error
// ⚠ recursion ลึก = จำนวน edge — edge เยอะ (2e5+) เสี่ยง stack overflow
// (กราฟไม่มีทิศทาง: ต้องลบ edge ฝั่งย้อนกลับด้วย — โค้ดนี้ใช้กับ directed ตรงๆ เท่านั้น)

// Sprague-Grundy (เกมทั่วไปที่ไม่ใช่หินธรรมดา): เปลี่ยนเกมเป็นเลข Grundy (G)
// G(state) = mex ของ { G(next_state) ทั้งหมดที่เดินไปถึงได้ }
// (mex = Minimum Excluded non-negative integer คือเลข 0,1,2... น้อยสุดที่ไม่มีในเซต)
int get_mex(unordered_set<int>& s) {
    int m = 0;
    while (s.count(m)) m++;
    return m;
}

2. Inclusion-Exclusion Principle (PIE) & Catalan Numbers

// PIE (นับรวม-หักออก): |A ∪ B ∪ C| = Σ|A| - Σ|A∩B| + |A∩B∩C|
// ใช้ Bitmask วนทุก subset (ถ้าจำนวนเซต ≤ 20):
ll total = 0;
for (int mask = 1; mask < (1 << k); mask++) {
    int bits = __builtin_popcount(mask);
    ll inter = calc_intersection(mask); // คำนวณขนาด intersection ของเซตใน mask
    if (bits % 2 == 1) total += inter;  // คี่บวก
    else total -= inter;                // คู่ลบ
}

// Catalan Numbers: C_0 = 1, C_n = (2n)! / ((n + 1)! * n!) = นับจำนวนวงเล็บ n คู่ที่ถูกต้อง,
// นับไบนารีทรี n โหนด, นับทางเดินจาก (0,0) ไป (n,n) ที่ไม่ข้ามเส้นทแยงมุม
ll catalan(ll n) {
    return nCr(2 * n, n) * inv(n + 1) % MOD;
}
